== Installation ==
run as root:
     echo "<?php echo " 17   * * * *  sudo -u www-data -- python3 /download_to_sql_fi.py ";?>" >> /etc/crontab

 Setup database MySQL table:
       CREATE USER ""@"%" identified by "";
       CREATE DATABASE 
       CREATE TABLE 

        GRANT ALL PRIVILEGES ON .* TO ";"@"%";

                                                                                       OR

        GRANT ALL PRIVILEGES ON . TO ";"@"%";
